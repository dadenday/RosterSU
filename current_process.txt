def process_sheet_v3(rows, alias_regex, date_str, filename, sheet_obj=None, context: Optional[ParseContext] = None, zone_blocks: list = None, cache: Optional[Dict] = None):
    """
    Main sheet processing entry point (wraps pure parsers with logging).
    
    Phase 2.5: Now supports explicit cache injection for pure function behavior.
    Pass cache={} for deterministic, thread-safe, parallel-safe operation.
    
    Side effects (only when cache is None):
    - Uses global _cell_norm_cache (cleared at start)
    
    For testing, use the pure functions instead:
    - parse_shift_sheet_pure() - for SHIFT sheets
    - parse_flight_sheet_pure() - for FLIGHT sheets
    
    Logic C: Zone Detection (Row-based)
    Logic: Ghost Buster implementation
    
    UPDATED: Now uses ParseContext.global_date as the single source of truth.
    The date_str parameter is deprecated - use context.global_date instead.
    
    NEW: v6.1 - Structural zone detection from merged ranges (if zone_blocks provided)
    Falls back to heuristic detection if zone_blocks is None or empty.
    
    Phase 2: Uses ParsedSheet and FlightRow dataclasses internally.
    Phase 6: Core logic is in pure functions; this is now PURE with explicit cache.
    """
    # Phase 2.5: Explicit cache injection for purity
    local_cache = cache if cache is not None else {}
    
    # Local wrapper for norm_cell that uses our cache - avoids modifying all call sites
    def norm_cell_cached(cell):
        return norm_cell(cell, cache=local_cache)

    # LAYER 1 COMPLIANCE: Use global_date from context if available
    # This ensures date is decided BEFORE parsing begins
    actual_date = context.global_date if context else date_str
    
    if actual_date == "Unknown":
        log_debug('DATE_REJECTED', {
            'reason': 'global_date_is_unknown',
            'filename': filename,
            'sheet': sheet_obj,
            'context_provided': context is not None
        })
        # Phase 2.5: Return ParsedSheet, not dict
        return ParsedSheet(date="Unknown", shift=None, flights=[], sheet_name=sheet_obj or "Unknown")
    
    # Phase 2: Use ParsedSheet dataclass internally
    result = ParsedSheet(
        date=actual_date,
        shift=None,
        flights=[],
        sheet_name=sheet_obj or "Unknown"
    )
    row_count = len(rows)
    processed_indices = set()

    # === v6.1: STRUCTURAL ZONE DETECTION ===
    # Use zone_blocks if provided, otherwise fall back to heuristics
    use_structural_zones = zone_blocks is not None and len(zone_blocks) > 0
    
    if use_structural_zones:
        log_debug('ZONE_MODE_STRUCTURAL', {
            'sheet': sheet_obj,
            'zone_blocks': len(zone_blocks)
        })

    current_zone = ""
    ZONE_SEQUENCE = ["SÂN ĐỖ", "BĂNG CHUYỀN", "TRẢ HÀNH LÝ"]
    zone_sequence_idx = 0
    last_start_hour = -1
    active_ca_type = None
    last_active_ca_type = None
    
    # v6.0: Count-based zone progression (fallback only)
    # Each zone should have at least MIN_ITEMS_PER_ZONE shift items before advancing
    MIN_ITEMS_PER_ZONE = 15
    zone_shift_counts = {"SÂN ĐỖ": 0, "BĂNG CHUYỀN": 0, "TRẢ HÀNH LÝ": 0}
    first_zone_marker_row = -1  # Row where first explicit SÂN ĐỖ marker found
    
    # 1. Statistical Header Discovery (v5.9)
    header_row_idx, col_mapping = find_header_mapping(rows)
    shift_col = col_mapping.get('shift')
    
    # 1.1 Sheet-level Identification for VS-BX (v5.9)
    sheet_header_text = ""
    for row in rows[:15]:
        sheet_header_text += " ".join([norm_cell_cached(c or "") for c in row[:50]]) + " "

    is_vsbx = "VỆ SINH" in sheet_header_text or "VS-BX" in sheet_header_text
    is_maca = sheet_obj and "MÃ CA" in norm_cell_cached(sheet_obj)
    # Header Mapping (v5.8) - Still used for Flight Zone strict filtering
    inside_cols = []
    outside_cols = []
    header_found = False

    for r_idx in range(row_count):
        if r_idx in processed_indices: continue
        row = rows[r_idx]
        items_curr = parse_row_items(row)
        
        # Optimization: Reduce per-row string churn
        cells_upper = [norm_cell_cached(x) for x in items_curr if x]

        def row_has(substr):
            return any(substr in cell for cell in cells_upper)
        
        # v6.2: Zone markers only appear in columns 0-1. Checking later columns
        # causes false positives (e.g., training notes mentioning "SÂN ĐỖ").
        # Zone marker columns: col 0 (SÂN ĐỔ), col 1 (TRẢ HÀNH LÝ ĐẾN)
        zone_cols_upper = [norm_cell_cached(x) for x in items_curr[:2] if x]
        
        def zone_cols_have(substr):
            """Check if zone marker appears in columns 0-1 only."""
            return any(substr in cell for cell in zone_cols_upper)
        
        # 2. Dynamic Header Detection (Once per sheet) - Legacy Flight Logic Support
        if not header_found and row_has("SERIAL") and row_has("CALLSIGN"):
            for c_idx, cell in enumerate(items_curr):
                c_val = norm_cell_cached(cell or "")
                if any(x in c_val for x in ["HL ĐI", "HÀNH LÝ ĐI", "TRẢ HL ĐẾN", "QUÁ KHỔ", "CHẤT XẾP HL"]):
                    inside_cols.append(c_idx)
                elif any(x in c_val for x in ["SÂN ĐỖ", "CẢNH GIỚI", "XI NHAN", "TÀU BAY"]):
                    outside_cols.append(c_idx)
            header_found = True
            continue 

        # 2. Zone Switching Logic
        # v6.1: Structural zones take priority over heuristics
        found_explicit_zone = False
        
        # === STRUCTURAL ZONE DETECTION (v6.1) ===
        if use_structural_zones:
            # Use merged range zone blocks - this is authoritative
            structural_zone = get_zone_for_row(r_idx, zone_blocks)
            if structural_zone:
                current_zone = structural_zone
                found_explicit_zone = True
                # Update zone_sequence_idx for consistency
                if structural_zone == "SÂN ĐỖ":
                    zone_sequence_idx = 0
                elif structural_zone == "BĂNG CHUYỀN":
                    zone_sequence_idx = 1
                elif structural_zone == "TRẢ HÀNH LÝ":
                    zone_sequence_idx = 2
        
        # === HEURISTIC ZONE DETECTION (fallback) ===
        # Only use heuristics if structural detection is not available or failed
        # v6.2: Use zone_cols_have() to only check columns 0-1 for zone markers
        if not use_structural_zones or not found_explicit_zone:
            # Explicit TRẢ HÀNH LÝ marker always wins (highest priority)
            if zone_cols_have("TRẢ HÀNH LÝ") or zone_cols_have("TRA HANH LY"):
                current_zone = "TRẢ HÀNH LÝ"
                zone_sequence_idx = 2
                found_explicit_zone = True
                last_start_hour = -1
                active_ca_type = None
                last_active_ca_type = None
            # Explicit SÂN ĐỖ marker - record first occurrence (also handles SÂN ĐỔ variant)
            elif zone_cols_have("SÂN ĐỖ") or zone_cols_have("SAN DO") or zone_cols_have("SÂN ĐỔ"):
                if first_zone_marker_row == -1:
                    first_zone_marker_row = r_idx
                current_zone = "SÂN ĐỖ"
                zone_sequence_idx = 0
                found_explicit_zone = True
                last_start_hour = -1
                active_ca_type = None
                last_active_ca_type = None
            # Explicit BĂNG CHUYỀN marker
            elif zone_cols_have("BĂNG CHUYỀN") or zone_cols_have("BANG CHUYEN"):
                current_zone = "BĂNG CHUYỀN"
                zone_sequence_idx = 1
                found_explicit_zone = True
                last_start_hour = -1
                active_ca_type = None
                last_active_ca_type = None

        # Identify current shift type (SANG/TOI) from row markers
        if row_has("CA SÁNG") or row_has("CA SANG") or row_has("CA 1"):
            active_ca_type = "SANG"
        elif row_has("CA TỐI") or row_has("CA TOI") or row_has("CA 2"):
            active_ca_type = "TOI"

        # 3. Row Parsing & Zone Tracking (v6.0: Count-based progression)
        # Identify if this is a shift row for zone tracking (ANY name)
        any_name_idx = -1
        for i, item in enumerate(items_curr):
            if is_valid_name_generic(item):
                any_name_idx = i
                break
